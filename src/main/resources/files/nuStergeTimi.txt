mvn compile
PENTRU CLIENT: mvn exec:java -Dexec.mainClass=ClientCommunication
PENTRU SERVER: mvn exec:java -Dexec.mainClass=ServerCommunication
PENTRU Creare de topics: mvn exec:java -Dexec.mainClass=CreateTopics

creat imagine docker: docker-compose -f docker-compose.yml up -d
